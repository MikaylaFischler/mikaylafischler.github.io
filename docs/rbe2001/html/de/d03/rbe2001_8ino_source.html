308 /*!
309  @brief Arduino loop function
310  @details Called until system exit. We delay it to not update too fast.
311 */
312 void loop() {
313  // keep track of gyro's position
314  gyro->maintainContinuousRotation();
315 
316  // only update LCD occasionally
317  if (lcd_timer > 50000) {
318  lcd_timer = 0;
319  lcd.clear();
320  uint16_t val = analogRead(A2);
321  lcd.print((float)val * (5.0/1023.0) * 2.0); // disp battery voltage
322  lcd.print("v Fault: ");
323  lcd.print(drive->getLeft()->getFault());
324  lcd.print(" ");
325  lcd.print(drive->getRight()->getFault());
326 
327  lcd.setCursor(0,1);
328 
329  rsensor->printMap(&lcd);
330  lcd.print(" ");
331  if (rsensor->onHorizontalLine()) { lcd.print("y"); } else { lcd.print("n"); }
332  if (rsensor->centered()) { lcd.print("y"); } else { lcd.print("n"); }
333  lcd.print(" ");
334  lcd.print(rsensor->getBias());
335  }
336 
337  // testing code
338  if (digitalRead(USER_BTN_0)) {
339  state = 1;
340  } else if (digitalRead(USER_BTN_1)) {
341  state = 0;
342  drive->stop();
343  arm->stop();
344  intake->stop();
345  }
346 
347  // update horizontal line global
348  horiz_line = rsensor->onHorizontalLine();
349 
350  // update LCD timer
351  lcd_timer += micros() - last_micros;
352 
353  // step through the virtually multithreaded state machine
354  if (state == 1) { t_robot->stepVirtualThreads(micros() - last_micros); }
355 
356  // keep track of elapsed time
357  last_micros = micros();
358 
359  // recieve data from bluetooth
360  messenger->step_recv();
361 }
