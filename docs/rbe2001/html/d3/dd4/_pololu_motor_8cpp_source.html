108  // 0 to 1023 for 0v to 5v

110 
111  // the next uncommented line is equivalent to the next commented line but has one less operation
112  // float voltage = (reading / 1023.0) * 5.0;0.004887585532746823069403714565
113  double voltage = ((double)reading / 1023.0) * 5.0;
114 
115  // convert to current (0.525 volts = 1.0 amps)
116  return voltage / 0.525;
117  } else { return 0; }
