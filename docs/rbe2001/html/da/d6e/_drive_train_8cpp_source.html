256  @brief Run drive PD with line follower PD for guidance.

258 void DriveTrain::driveLineGuided(void) {
259  // start PD controllers
260  if (!pd_running) {
261  pd_left->SetMode(AUTOMATIC);
262  pd_right->SetMode(AUTOMATIC);
263  pd_running = true;
264  }
265 
266  if (!pd_line_running) {
267  pd_line->SetMode(AUTOMATIC);
268  pd_line_running = true;
269  }
270 
271  // set input positions
272  pd_input_l = getLeftPosition();
273  pd_input_r = getRightPosition();
274  pd_input_line = rsensor->getBias();
275 
276  // calculate PD outputs
277  pd_left->Compute();
278  pd_right->Compute();
279  pd_line->Compute();
280 
281  // write PD outputs
282  // this is drive straight so just check one side
283  if (pd_output_l > 0) {
284  if (pd_output_line > 0) {
285  tankDrive(pd_output_l, pd_output_r + pd_output_line);
286  } else {
287  tankDrive(pd_output_l - pd_output_line, pd_output_r);
288  }
289  } else {
290  if (pd_output_line > 0) {
291  tankDrive(pd_output_l, pd_output_r - pd_output_line);
292  } else {
293  tankDrive(pd_output_l + pd_output_line, pd_output_r);
294  }
295  }
296 }

300  @param[in] tolerance_l Left tolerance in inches

304 bool DriveTrain::nearSetpoints(double tolerance_l, double tolerance_r) {
305  bool leftNear, rightNear;
