112  radiation = 0xFF;
113  }
114 
115  Packet* rad_pkt = new Packet(MSG_T_RADIATION, TEAM_NUMBER, 0x00, radiation);
116  bt_cli->send(rad_pkt);
117  if (this->debug) {
118  Serial.print("Team 16 sent radiation data of ");
119  Serial.println(radiation, HEX);
120  }
121 
122  this->next_radiation_send = 2; // once every 3 seconds
